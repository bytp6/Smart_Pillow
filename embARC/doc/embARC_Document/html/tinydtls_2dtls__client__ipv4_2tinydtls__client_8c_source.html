136 static int get_psk_info(struct dtls_context_t *ctx, const session_t *session,
137  dtls_credentials_type_t type,
138  const unsigned char *id, size_t id_len,
139  unsigned char *result, size_t result_length)
140 {
141  struct keymap_t {
142  unsigned char *id;
143  size_t id_length;
144  unsigned char *key;
145  size_t key_length;
146  } psk = { (unsigned char *)"Client_identity", 15, (unsigned char *)"secretPSK", 9 };
147 
148  switch (type) {
149  case DTLS_PSK_IDENTITY:
150  if (id_len) {
151  dtls_debug("got psk_identity_hint: '%.*s'\n", id_len, id);
152  }
153  if (result_length < psk.id_length) {
154  dtls_warn("cannot set psk_identity -- buffer too small\n");
155  return dtls_alert_fatal_create(DTLS_ALERT_INTERNAL_ERROR);
156  }
157  memcpy(result, psk.id, psk.id_length);
158  return psk.id_length;
159  case DTLS_PSK_KEY:
160  if (id_len != psk.id_length || memcmp(psk.id, id, id_len) != 0) {
161  dtls_warn("PSK for unknown id requested, exiting\n");
162  return dtls_alert_fatal_create(DTLS_ALERT_ILLEGAL_PARAMETER);
163  } else if (result_length < psk.key_length) {
164  dtls_warn("cannot set psk -- buffer too small\n");
165  return dtls_alert_fatal_create(DTLS_ALERT_INTERNAL_ERROR);
166  }
167  memcpy(result, psk.key, psk.key_length);
168  return psk.key_length;
169  default:
170  dtls_warn("unsupported request type: %d\n", type);
171  break;
172  }
173  return dtls_alert_fatal_create(DTLS_ALERT_INTERNAL_ERROR);
174 }

177 int dtls_event_handler(struct dtls_context_t *ctx, session_t *session,
178  dtls_alert_level_t level, unsigned short code)
179 {
180  unsigned int event;
181 
182  if (code == DTLS_EVENT_CONNECTED) {
183  EMBARC_PRINTF("connected to server\r\n");
184  event = DTLS_EVENT_CONNECTED;
185  }
186 
187  if (level == DTLS_ALERT_LEVEL_WARNING) {
188  EMBARC_PRINTF("warning alert from peer, code:%d\r\n", code);
189  event = DTLS_ALERT_LEVEL_WARNING;
190  } else if (level == DTLS_ALERT_LEVEL_FATAL) {
191  EMBARC_PRINTF("fatal alert from peer, code:%d\r\n", code);
192  event = DTLS_ALERT_LEVEL_FATAL;
193  }
194 
195  process_post(PROCESS_BROADCAST, dtls_event, (process_data_t)event);
196  return 0;
197 }

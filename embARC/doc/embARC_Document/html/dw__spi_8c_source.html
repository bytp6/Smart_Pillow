397 static int32_t dw_spi_dis_cbr(DEV_SPI_INFO *spi_info_ptr, uint32_t cbrtn)
398 {
399  DW_SPI_CTRL *spi_ctrl_ptr = (DW_SPI_CTRL *)(spi_info_ptr->spi_ctrl);
400  int32_t ercd = E_OK;
401 
402  if ((spi_info_ptr->status & DW_SPI_IN_XFER) != 0) { /* only in transfer need do check */
403  switch (cbrtn) {
404  case DW_SPI_RDY_SND:
405  DW_SPI_CHECK_EXP((spi_info_ptr->status & DW_SPI_IN_XFER) == DW_SPI_IN_TX, E_CTX);
406  spi_info_ptr->status &= ~(DW_SPI_IN_TX);
407  break;
408  case DW_SPI_RDY_RCV:
409  DW_SPI_CHECK_EXP((spi_info_ptr->status & DW_SPI_IN_XFER) == DW_SPI_IN_RX, E_CTX);
410  spi_info_ptr->status &= ~(DW_SPI_IN_RX);
411  break;
412  case DW_SPI_RDY_XFER:
413  DW_SPI_CHECK_EXP((spi_info_ptr->status & DW_SPI_IN_XFER) == DW_SPI_IN_XFER, E_CTX);
414  spi_info_ptr->status &= ~(DW_SPI_IN_XFER);
415  break;
416  default:
417  break;
418  }
419  }
420 
421  dw_spi_mask_interrupt(spi_ctrl_ptr->dw_spi_regs, DW_SPI_IMR_XFER);
422 
423  if (spi_ctrl_ptr->int_status & DW_SPI_GINT_ENABLE) {
424  int_disable(spi_ctrl_ptr->intno);
425  spi_ctrl_ptr->int_status &= ~DW_SPI_GINT_ENABLE;
426  }
427 
428 error_exit:
429  return ercd;
430 }

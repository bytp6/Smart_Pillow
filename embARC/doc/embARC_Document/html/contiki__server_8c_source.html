261 PROCESS_THREAD(dhcpc_process, ev, data)
262 {
263  DEV_WNIC_PTR wifi;
264  PROCESS_BEGIN();
265 
266  PRINTF("\r\nNow trying to connect to WIFI hotspot, Please wait about 30s!\r\n");
267  /* wait wifi connected */
268  while (1) {
269  PROCESS_WAIT_EVENT();
270  if (ev == wnic_event) {
271  wifi = (DEV_WNIC_PTR)data;
272  if (wifi->wnic_info.conn_status == WNIC_CONNECTED) {
273  break;
274  }
275  }
276  }
277  PRINTF("Now start get ip address using DHCP, Please wait about 30s!\r\n");
278 
279  dhcpc_init(uip_lladdr.addr, sizeof(uip_lladdr.addr));
280  dhcpc_request();
281 
282  while(1) {
283  PROCESS_WAIT_EVENT();
284  if (ev == PROCESS_EVENT_EXIT) {
285  process_exit(&dhcpc_process);

287  dhcpc_appcall(ev, data);
288  }

290  PROCESS_END();

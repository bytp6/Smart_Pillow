135 int32_t exc_entry_install(const uint32_t excno, EXC_ENTRY entry)
136 {
137  uint32_t status;
138 
139  EXC_ENTRY *table = (EXC_ENTRY *)_arc_aux_read(AUX_INT_VECT_BASE);
140 
141  if (excno < NUM_EXC_ALL && entry != NULL
142  && table[excno] != entry) {
143  status = cpu_lock_save();
144  /* directly write to mem, as arc gets exception handler from mem not from cache */
145  /* FIXME, here maybe icache is dirty, need to be invalidated */
146  table[excno] = entry;
147 
148  if (_arc_aux_read(AUX_BCR_D_CACHE) > 0x2) {
149  /* dcache is available */
150  dcache_flush_line((uint32_t)&table[excno]);
151  }
152 
153  if (_arc_aux_read(AUX_BCR_D_CACHE) > 0x2) {
154  /* icache is available */
155  icache_invalidate_line((uint32_t)&table[excno]);
156  }
157  cpu_unlock_restore(status);
158  return 0;
159  }
160  return -1;
161 }

212 void exc_int_init(void)
213 {
214  uint32_t i;
215  uint32_t status;
216  struct aux_irq_ctrl {
217  /* note: little endian */
218  uint32_t save_nr_gpr_pairs:5, res:4,
219  save_blink:1, save_lp_regs:1, save_u_to_u:1,
220  res2:1, save_idx_regs:1, res3:18;
221  } ictrl;
222 #ifndef ARC_FEATURE_RF16
223  ictrl.save_nr_gpr_pairs = 6; /* r0 to r11 (r12 saved manually) */
224 #else
225  ictrl.save_nr_gpr_pairs = 3; /* r0 to r3, r10, r11 */
226 #endif
227  ictrl.save_blink = 1;
228  ictrl.save_lp_regs = 1; /* LP_COUNT, LP_START, LP_END */
229  ictrl.save_u_to_u = 0; /* user ctxt saved on kernel stack */
230  ictrl.save_idx_regs = 1; /* JLI, LDI, EI */
231 
232  status = cpu_lock_save();
233  for (i = NUM_EXC_CPU; i < NUM_EXC_ALL; i++) {
234  /* interrupt level triggered, disabled, priority is the lowest */
235  _arc_aux_write(AUX_IRQ_SELECT, i);
236  _arc_aux_write(AUX_IRQ_ENABLE, 0);
237  _arc_aux_write(AUX_IRQ_TRIGGER, 0);
238  _arc_aux_write(AUX_IRQ_PRIORITY, INT_PRI_MAX - INT_PRI_MIN);
239  }
240  _arc_aux_write(AUX_IRQ_CTRL, *(uint32_t *)&ictrl);
241 
242  cpu_unlock_restore(status);
243 
245  int_ipm_set(INT_PRI_MAX);
246 }

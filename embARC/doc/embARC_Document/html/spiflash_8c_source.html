92 Inline int32_t spi_send_cmd(DEV_SPI_TRANSFER *xfer)
93 {
94  uint32_t cpu_status;
95  int32_t ercd = 0;
96 
97  cpu_status = cpu_lock_save();
98 
99  /* select device */
100  ercd = spi_flash->spi_control(SPI_CMD_MST_SEL_DEV, CONV2VOID(cs_flash));
101  SPIFLASH_CHECK_EXP(ercd == E_OK, -1);
102  ercd = spi_flash->spi_control(SPI_CMD_TRANSFER_POLLING, CONV2VOID(xfer));
103  /* deselect device */
104  spi_flash->spi_control(SPI_CMD_MST_DSEL_DEV, CONV2VOID(cs_flash));
105  SPIFLASH_CHECK_EXP(ercd == E_OK, -1);
106 
107  cpu_unlock_restore(cpu_status);
108  if (ercd == E_OK) ercd = 0;
109 
110 error_exit:
111  return ercd;
112 }

138 uint32_t flash_read_id(void)
139 {
140  uint32_t id = 0;
141  uint8_t local_buf[4];
142  DEV_SPI_TRANSFER cmd_xfer;
143 
144  local_buf[0] = RDID;
145 
146  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 1);
147  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, local_buf, 1, 3);
148  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
149 
150  if (spi_send_cmd(&cmd_xfer) == 0) {
151  id = (local_buf[0] << 16) | (local_buf[1] << 8) | local_buf[2];
152  } else {
153  id = SPI_FLASH_NOT_VALID;
154  }
155 
156  return id;

235 int32_t flash_write_enable(void)
236 {
237  uint8_t local_buf[3];
238  DEV_SPI_TRANSFER cmd_xfer;
239 
240  uint32_t status = 0;
241  do {
242  local_buf[0] = WREN;
243 
244  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 1);
245  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
246  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
247 
248  if (spi_send_cmd(&cmd_xfer) != 0) {
249  return -1;
250  }
251  status = flash_read_status();
252  if (status == SPI_FLASH_NOT_VALID) {
253  return -1;
254  }
255  // clear protection bits
256  // Write Protect. and Write Enable.
257  if( (status & 0xfc) && (status & 0x02) ) {
258  local_buf[0] = WRSR; // write status
259  local_buf[1] = 0x00; // write status
260  local_buf[2] = 0x00; // write status
261 
262  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 3);
263  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
264  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
265 
266  if (spi_send_cmd(&cmd_xfer) != 0) {
267  return -1;
268  }
269  status = 0;
270  }
271  } while ( status != 0x02);
272 
273  return 0;
274 }

286 int32_t flash_erase(uint32_t address, uint32_t size)
287 {
288  uint32_t last_address;
289  uint32_t count = 0;
290  uint8_t local_buf[4];
291  DEV_SPI_TRANSFER cmd_xfer;
292 
293  // start address of last sector
294  last_address = (address + size) & (~(FLASH_SECTOR_SIZE - 1));
295 
296  // start address of first sector
297  address &= ~(FLASH_SECTOR_SIZE - 1);
298 
299  do {
300  if (flash_write_enable() != 0) {
301  return -1;
302  }
303 
304  local_buf[0] = SE;
305  local_buf[1] = (address >> 16) & 0xff;
306  local_buf[2] = (address >> 8) & 0xff;
307  local_buf[3] = address & 0xff;
308 
309  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 4);
310  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
311  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
312 
313  if (spi_send_cmd(&cmd_xfer) != 0) {
314  return -1;
315  }
316 
317  if (flash_wait_ready() != 0) {
318  return -1;
319  }
320 
321  address += FLASH_SECTOR_SIZE;
322  count++;
323  } while(address <= last_address);
324 
325  return (int32_t)count;
326 }

338 int32_t flash_write(uint32_t address, uint32_t size, const void *data)
339 {
340 
341  uint8_t local_buf[4];
342  DEV_SPI_TRANSFER cmd_xfer;
343  DEV_SPI_TRANSFER data_xfer;
344 
345  uint32_t first = 0;
346  uint32_t size_orig = size;
347 
348  if (flash_wait_ready() != 0) {
349  return -1;
350  }
351 
352  first = FLASH_PAGE_SIZE - (address & (FLASH_PAGE_SIZE - 1));
353 
354  do {
355  // send write enable command to flash
356  if (flash_write_enable() != 0) {
357  return -1;
358  }
359  if (flash_wait_ready() != 0) {
360  return -1;
361  }
362 
363  first = first < size ? first : size;
364 
365  local_buf[0] = PP;
366  local_buf[1] = (address >> 16) & 0xff;
367  local_buf[2] = (address >> 8) & 0xff;
368  local_buf[3] = address & 0xff;
369 
370  DEV_SPI_XFER_SET_TXBUF(&data_xfer, data, 0, first);
371  DEV_SPI_XFER_SET_RXBUF(&data_xfer, NULL, 0, 0);
372  DEV_SPI_XFER_SET_NEXT(&data_xfer, NULL);
373 
374  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 4);
375  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
376  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, &data_xfer);
377 
378  if (spi_send_cmd(&cmd_xfer) != 0) {
379  return -1;
380  }
381 
382  size -= first;
383  address += first;
384  data += first;
385  first = FLASH_PAGE_SIZE;
386 
387  } while (size);
388 
389  if (flash_wait_ready() != 0) {
390  return -1;
391  }
392 
393  return (int32_t)(size_orig);
394 }

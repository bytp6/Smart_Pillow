158

163 
164  if (!strcmp(argv[1], "challenge")) {
165  EMBARC_PRINTF("try to steal the secret\r\n");
166 
167  switch (argv[2][0]) {
168  case '0' :
169  EMBARC_PRINTF("the secret from container 1 context is:%s\r\n", challenge->secret);
170  break;
171  case '1' :
172  EMBARC_PRINTF("the secret init data is:%s\r\n", secret_const_data);
173  break;
174  case '2' :
175  EMBARC_PRINTF("the private shared data is:%s\r\n", private_shared_data);
176  break;
177  default:
178  EMBARC_PRINTF("Please use challenge 0, 1, or 2\r\n");
179  return E_OK;
180  }
181  return E_OK;
182  }
183 
184  opterr = 0;
185  optind = 1;
186 
187  while ((opt=getopt(argc, argv, "tp:s:n:hH?")) != -1) {
188  switch (opt) {
189  case 'h':
190  case '?':
191  case 'H':
192  main_help();
193  goto error_exit;
194  break;
195  case 'p':
196  pwd = optarg;
197  break;
198  case 's':
199  secret = optarg;
200  if (strlen(secret) > SECRET_LEN) {
201  EMBARC_PRINTF("secret is too long\r\n");
202  goto error_exit;
203  }
204  break;
205  case 't':
206  EMBARC_PRINTF("Requesting operation from trusted container\r\n");
207  container_call(container2, trusted_ops);
208  return E_OK;
209  break;
210  case 'n':
211  new_pwd = optarg;
212  if (strlen(new_pwd) > PWD_LEN) {
213  EMBARC_PRINTF("max password length 6 bytes\r\n");
214  goto error_exit;
215  }
216  break;
217  default:
218  main_help();
219  goto error_exit;
220  break;
221  }
222  }
223 
224  /* UART is in background container, so a hacker could hijack or access the UART and public_data to
225  get the secret. Here it's just for demo. In a real applicaton, container 2's method should be used */
226  if (pwd != NULL) {
227  if (secret == NULL && new_pwd == NULL) {
228  if(container_call(container1, operate_secret, pwd, GET_SECRET, public_data)) {
229  EMBARC_PRINTF("get secret failed\r\n");
230  } else {
231  EMBARC_PRINTF("the secret is:%s\r\n", public_data);
232  memset(public_data, 0, SECRET_LEN);
233  }
234  } else if (secret != NULL && new_pwd == NULL) {
235  if(container_call(container1, operate_secret, pwd, SET_SECRET, secret)) {
236  EMBARC_PRINTF("set new secret failed\r\n");
237  }
238  } else if (secret == NULL && new_pwd != NULL) {
239  if(container_call(container1, operate_secret, pwd, SET_PWD, new_pwd)) {
240  EMBARC_PRINTF("set new password failed\r\n");
241  }
242  } else {
243  EMBARC_PRINTF("cannot set password and secret at the same time\r\n");
244  }
245  } else {
246  EMBARC_PRINTF("no password input\r\n");
247  }
248 
249 error_exit:
250  return E_OK;
251 }

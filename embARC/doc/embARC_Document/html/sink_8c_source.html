110 static uip_ds6_maddr_t * join_mcast_group(void)
111 {
112  uip_ipaddr_t addr;
113  uip_ds6_maddr_t *rv;
114 
115  /* First, set our v6 global */
116  uip_ip6addr(&addr, 0xaaaa, 0, 0, 0, 0, 0, 0, 0);
117  uip_ds6_set_addr_iid(&addr, &uip_lladdr);
118  uip_ds6_addr_add(&addr, 0, ADDR_AUTOCONF);
119 
120  /*
121  * IPHC will use stateless multicast compression for this destination
122  * (M=1, DAC=0), with 32 inline bits (1E 89 AB CD)
123  */
124  uip_ip6addr(&addr, 0xFF1E,0,0,0,0,0,0x89,0xABCD);
125  rv = uip_ds6_maddr_add(&addr);
126 
127  if(rv) {
128  EMBARC_PRINTF("Joined multicast group ");
129  PRINT6ADDR(&uip_ds6_maddr_lookup(&addr)->ipaddr);
130  EMBARC_PRINTF("\r\n");
131  }
132  return rv;
133 }

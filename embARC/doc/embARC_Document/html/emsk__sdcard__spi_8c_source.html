133 static int32_t sdcard_spi_0_spi_cs(int32_t cs)
134 {
135  int32_t ercd = E_OK;
136  DEV_SPI *sdcard_spi_ptr;
137  uint32_t cs_line = EMSK_SDCARD_SPI_0_LINE;
138 #ifdef EMSK_SDCARD_SPI_0_CPULOCK_ENABLE
139  static unsigned int cs_cpu_status;
140 #endif
141 
142  sdcard_spi_ptr = spi_get_dev(EMSK_SDCARD_SPI_0_ID);
143  EMSK_SDCARD_SPI_CHECK_EXP(sdcard_spi_ptr != NULL, E_OBJ);
144 
145  if (cs == SDCARD_SPI_SELECTED) {
146 #ifdef EMSK_SDCARD_SPI_0_CPULOCK_ENABLE
147  cs_cpu_status = cpu_lock_save();
148 #endif
149  ercd = sdcard_spi_ptr->spi_control(SPI_CMD_MST_SEL_DEV, CONV2VOID(cs_line));
150  } else {
151  ercd = sdcard_spi_ptr->spi_control(SPI_CMD_MST_DSEL_DEV, CONV2VOID(cs_line));
152 #ifdef EMSK_SDCARD_SPI_0_CPULOCK_ENABLE
153  cpu_unlock_restore(cs_cpu_status);
154 #endif
155  }
156 
157 error_exit:
158  return ercd;
159 }

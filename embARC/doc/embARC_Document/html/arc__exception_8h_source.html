206 Inline int32_t int_ipm_get(void)
207 {
208  uint32_t status;
209  status = _arc_aux_read(AUX_STATUS32);
210  status = (status >> 1) & 0xf;

212  return ((int32_t)status + INT_PRI_MIN);

220 Inline void int_ipm_set(int intpri)
221 {
222  volatile uint32_t status;
223 
224  intpri = intpri - INT_PRI_MIN;
225 
226  status = _arc_aux_read(AUX_STATUS32) & ~0x1e;
227 
228  status = status | ((intpri << 1) & 0x1e);
229  /* sr cannot write AUX_STATUS32 */
230  Asm("kflag %0"::"ir"(status));

272 static int get_psk_info(struct dtls_context_t *ctx, const session_t *session,
273  dtls_credentials_type_t type,
274  const unsigned char *id, size_t id_len,
275  unsigned char *result, size_t result_length)
276 {
277  switch (type) {
278  case DTLS_PSK_IDENTITY:
279  if (result_length < psk.id_length) {
280  EMBARC_PRINTF("cannot set psk_identity -- buffer too small\n");
281  return dtls_alert_fatal_create(DTLS_ALERT_INTERNAL_ERROR);
282  }
283  memcpy(result, psk.id, psk.id_length);
284  return psk.id_length;
285  case DTLS_PSK_KEY:
286  if (id_len != psk.id_length || memcmp(psk.id, id, id_len) != 0) {
287  EMBARC_PRINTF("PSK for unknown id requested, exiting\n");
288  return dtls_alert_fatal_create(DTLS_ALERT_ILLEGAL_PARAMETER);
289  } else if (result_length < psk.key_length) {
290  EMBARC_PRINTF("cannot set psk -- buffer too small\n");
291  return dtls_alert_fatal_create(DTLS_ALERT_INTERNAL_ERROR);
292  }
293  memcpy(result, psk.key, psk.key_length);
294  return psk.key_length;
295  default:
296  EMBARC_PRINTF("unsupported request type: %d\n", type);
297  break;
298  }
299  return dtls_alert_fatal_create(DTLS_ALERT_INTERNAL_ERROR);
300 }

303 static int dtls_event_handler(struct dtls_context_t *ctx, session_t *session,
304  dtls_alert_level_t level, unsigned short code)
305 {
306  unsigned int event;
307 
308  if (code == DTLS_EVENT_CONNECTED) {
309  EMBARC_PRINTF("connected to server\r\n");
310  event = DTLS_EVENT_CONNECTED;
311  }
312 
313  if (level == DTLS_ALERT_LEVEL_WARNING) {
314  EMBARC_PRINTF("warning alert from peer, code:%d\r\n", code);
315  event = DTLS_ALERT_LEVEL_WARNING;
316  } else if (level == DTLS_ALERT_LEVEL_FATAL) {
317  EMBARC_PRINTF("fatal alert from peer, code:%d\r\n", code);
318  event = DTLS_ALERT_LEVEL_FATAL;
319  }
320 
321  process_post(PROCESS_BROADCAST, dtls_event, (process_data_t)event);
322  return 0;
323 }

353 PROCESS_THREAD(udp_client_process, ev, data)
354 {
355  DEV_WNIC_PTR wifi;
356  uip_ipaddr_t ip;
357  int ret;
358 
359  PROCESS_BEGIN();
360 
361 
362  process_start(&tcpip_process, NULL);
363  process_start(&mrf24g_process, NULL);
364  process_start(&try_send_process, NULL);
365 
366  EMBARC_PRINTF("---tinydtls client with secureshield example---\r\n");
367  EMBARC_PRINTF("\r\nNow trying to connect to WIFI Hotspot, Please wait about 30s!\r\n");
368  /* wait wifi connected */
369  while (1) {
370  PROCESS_WAIT_EVENT_UNTIL(ev == wnic_event);
371  wifi = (DEV_WNIC_PTR)data;
372  if (wifi->wnic_info.conn_status == WNIC_CONNECTED) {
373  break;
374  }
375  }
376 
377  /* set network parameters */
378  EMBARC_PRINTF("\r\n-----Set client node with new static ip address-----\r\n");
379  uiplib_ipaddrconv(LOCALHOST, &ip);
380  uip_sethostaddr(&ip);
381  ip_addr_dump("ip:", &ip);
382  uiplib_ipaddrconv(NETMASK, &ip);
383  uip_setnetmask(&ip);
384  ip_addr_dump("netmask:", &ip);
385  uiplib_ipaddrconv(NET_GATEWAY, &ip);
386  uip_setdraddr(&ip);
387  ip_addr_dump("gateway:", &ip);
388  EMBARC_PRINTF("\r\n\r\n");
389 
390  ret = container_call(container_tinydtls, interface_init_dtls, &server);
391  if (ret < 0) {
392  EMBARC_PRINTF("failed to init tinydtls\r\n");
393  PROCESS_EXIT();
394  }
395 
396  ret = container_call(container_tinydtls, interface_dtls_connect, dtls_context, &server);
397 
398  if (ret <= 0) {
399  EMBARC_PRINTF("failed to connect to server\r\n");
400  PROCESS_EXIT();
401  }
402  dtls_event = process_alloc_event();
403 
404  /* handle input */
405  while(1) {
406  PROCESS_WAIT_EVENT_UNTIL(ev == tcpip_event);
407  dtls_handle_read(dtls_context);
408  }
409 
410  PROCESS_END();
411 }

414 PROCESS_THREAD(try_send_process, ev, data)
415 {
416  static int i = 0;
417  PROCESS_BEGIN();
418  int send_status = 0;
419  static struct etimer et;
420 
421  /* wait for dtls connected */
422  while(1) {
423  PROCESS_WAIT_EVENT_UNTIL(ev == dtls_event);
424  if ((uint32_t)data == DTLS_EVENT_CONNECTED) {
425  break;
426  }
427  }
428 
429  /* every SEND_PERIOD seconds, send a message */
430  etimer_set(&et, SEND_PERIOD * CLOCK_SECOND);
431 
432  while(1) {
433  PROCESS_WAIT_EVENT();
434 
435  if (ev == dtls_event) {
436  if ((uint32_t)data == DTLS_ALERT_LEVEL_FATAL) {
437  PROCESS_EXIT();
438  }
439  }
440  if(ev == PROCESS_EVENT_TIMER) {
441  etimer_set(&et, SEND_PERIOD * CLOCK_SECOND); /* reset timer */
442 
443  sprintf(buf, "hello %d", i++);
444  send_status = strlen(buf);
445  send_status = container_call(container_tinydtls, interface_dtls_write,
446  dtls_context, &server, buf, send_status);
447 
448  if(send_status < 0 ) {
449  EMBARC_PRINTF("client send error %d\r\n", send_status);
450  }
451  }
452  }
453 
454  PROCESS_END();
455 }

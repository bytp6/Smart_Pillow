123 static unsigned int perf_end(void)
124 {
125  unsigned int end = 0;
126 
127  if (timer_current(perf_id, (void *)(&end)) < 0) {
128  return 0;
129  }
130 
131  if (start < end) {
132  return (end - start);
133  } else {
134  return (0xFFFFFFFF - start + end);
135  }
136 }

203  }

216 int32_t temp_sensor_read(int32_t *val)
217 {
218  int32_t ercd = E_OK;
219  uint8_t data[2];
220  int32_t temp;
221 
222  emsk_temp_sensor = iic_get_dev(BOARD_TEMP_SENSOR_IIC_ID);
223 
224  EMSK_TEMP_SENSOR_CHECK_EXP_NORTN(emsk_temp_sensor!=NULL);
225  EMSK_TEMP_SENSOR_CHECK_EXP_NORTN(val!=NULL);
226 
227  ercd = adt7420_reg_read(ADT7420_REG_TEMP_MSB, &data[0]);
228  ercd += adt7420_reg_read(ADT7420_REG_TEMP_LSB, &data[1]);
229 
230  if (ercd != 2) {
231  ercd = E_OBJ;
232  } else {
233  ercd = E_OK;
234  temp = ((int32_t)data[0] << 8) + ((int32_t)data[1]);
235  /* conversion (val = val*0.0625) in Degrees */
236  if (temp_resolution == ADT7420_RESOLUTION_13BIT) {
237  temp >>= 3;
238  if (temp & (1<<ADT7420_RESOLUTION_13BIT)) { /* negative temp */
239  temp = ((temp-8192) * 10) >> 4;
240  } else { /* positive temp */
241  temp = (temp * 10) >> 4;
242  }
243  } else {
244  if (temp & (1<<ADT7420_RESOLUTION_16BIT)) { /* negative temp */
245  temp = ((temp-65536) * 10) >> 7;
246  } else { /* positive temp */
247  temp = (temp * 10) >> 7;
248  }
249  }
250  *val = temp;
251  }
252 
253 error_exit:
254  return ercd;
255 }

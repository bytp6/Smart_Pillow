136 // Note that some parameters of this function are pointers. Care should be taken that Container 1 has access to the memory pointed at
137 static int32_t operate_secret(char* pwd, uint32_t cmd, char * data)
138 {
139  SECRET_CONTEXT *ctx;
140  uint32_t len;
141 
142  ctx = (SECRET_CONTEXT *)secureshield_container_ctx;
143 
144  if (ctx->initialized == 0) {
145  return -1;
146  }
147 
148  // protect against brute force password guessing; for better protection, number of fails should be stored persistently.
149  if (ctx->fails++ > 3) {
150  EMBARC_PRINTF("Too many failed attempts to obtain secret.\r\n");
151  // optionally, the secret could be erased here
152  return -1;
153  }
154 
155  // Check if caller is authorized to access the secret
156  if (secureshield_container_id_caller() == 0) {
157  if (strcmp(pwd, (const char *)ctx->pwd) != 0) {
158  return -1;
159  }
160  memset(pwd, 0, strlen(pwd));
161  } else if (secureshield_container_id_caller() == 2) {
162  EMBARC_PRINTF("container2 is trusted, no need to check password\r\n");
163  } else {
164  EMBARC_PRINTF("Unknown container tried to access the secret\r\n");
165  return -1;
166  }
167 
168  // Caller is trusted now, perform the requested operation
169  ctx->fails = 0;
170  if (cmd == GET_SECRET) {
171  // note that data is copied with Container 1 privileges to the memory pointer to by the 'data' argument
172  // this could be a security risk, but since the caller is trusted the data pointer is trusted as well
173  memcpy(data, ctx->secret, SECRET_LEN);
174  } else if (cmd == SET_SECRET) {
175  memset(ctx->secret, 0, SECRET_LEN);
176  len = strlen(data);
177  if (len > SECRET_LEN) {
178  len = SECRET_LEN;
179  }
180  memcpy(ctx->secret, data, len);
181  memset(data, 0, len);
182  } else if (cmd == SET_PWD) {
183  memset(ctx->pwd, 0, PWD_LEN);
184  len = strlen(data);
185  if (len > PWD_LEN) {
186  len = PWD_LEN;
187  }
188  memcpy(ctx->pwd, data, len);
189  memset(data, 0, len);
190  } else {
191  return -1;
192  }
193 
194  return 0;
195 }

199 static void trusted_ops(void)
200 {
201  uint8_t data[SECRET_LEN];
202  SECRET_CONTEXT2 *ctx;
203 
204  ctx = (SECRET_CONTEXT2 *)secureshield_container_ctx;
205 
206  EMBARC_PRINTF("container 2 is trusted\r\n");
207  /* data is on container's stack, normally other containers can not access it. As
208  container1 is a secure container, it can access container 2's stack */
209  if(container_call(container1, operate_secret, NULL, GET_SECRET, data)) {
210  EMBARC_PRINTF("get secret failed - someone tried to hack the system?\r\n");
211  return;
212  }
213 
214  if (! ctx->initialized) {
215  EMBARC_PRINTF("container 2: got the secret for the first time and stored it for private use\r\n");
216  memcpy(ctx->secret, data, SECRET_LEN);
217  ctx->initialized = 1;
218  } else if (strcmp((const char *)data, (const char *)ctx->secret) != 0) {
219  EMBARC_PRINTF("container 2: the secret changed, updated my private copy!\r\n");
220  memcpy(ctx->secret, data, SECRET_LEN);
221  } else {
222  EMBARC_PRINTF("got the secret, but not telling you since its secret and for use within this container only...\r\n");
223  }

225 }

278 int main(int argc, char **argv)
279 {
280  int32_t opt;
281  char *pwd = NULL;
282  char *secret = NULL;
283  char *new_pwd = NULL;
284 
285  if (container_call(container1, init_secret)) {
286  EMBARC_PRINTF("secret key initialized error\r\n");
287  return E_SYS;
288  }
289 
290  if (!strcmp(argv[1], "challenge")) {
291  EMBARC_PRINTF("try to steal the secret\r\n");
292  switch (argv[2][0]) {
293  case '0' :
294  EMBARC_PRINTF("the secret from container 1 context is:%s\r\n", challenge->secret);
295  break;
296  case '1' :
297  EMBARC_PRINTF("the secret from direct memory read is:%s\r\n", container1_secret);
298  break;
299  case '2' :
300  EMBARC_PRINTF("the password from direct memory read is:%s\r\n", container1_pwd);
301  break;
302  default:
303  EMBARC_PRINTF("Please use challenge 0, 1, or 2\r\n");
304  return E_OK;
305  }
306  return E_OK;
307  }
308 
309  opterr = 0;
310  optind = 1;
311 
312  while ((opt=getopt(argc, argv, "tp:s:n:hH?")) != -1) {
313  switch (opt) {
314  case 'h':
315  case '?':
316  case 'H':
317  main_help();
318  goto error_exit;
319  break;
320  case 'p':
321  pwd = optarg;
322  break;
323  case 's':
324  secret = optarg;
325  if (strlen(secret) > SECRET_LEN) {
326  EMBARC_PRINTF("secret is too long\r\n");
327  goto error_exit;
328  }
329  break;
330  case 't':
331  EMBARC_PRINTF("Requesting operation from trusted container\r\n");
332  container_call(container2, trusted_ops);
333  return E_OK;
334  break;
335  case 'n':
336  new_pwd = optarg;
337  if (strlen(new_pwd) > PWD_LEN) {
338  EMBARC_PRINTF("max password length 6 bytes\r\n");
339  goto error_exit;
340  }
341  break;
342  default:
343  main_help();
344  goto error_exit;
345  break;
346  }
347  }
348 
349  /* UART is in background container, so a hacker could hijack or access the UART and public_data to
350  get the secret. Here it's just for demo. In a real applicaton, container 2's method should be used */
351  if (pwd != NULL) {
352  if (secret == NULL && new_pwd == NULL) {
353  if(container_call(container1, operate_secret, pwd, GET_SECRET, public_data)) {
354  EMBARC_PRINTF("get secret failed\r\n");
355  } else {
356  EMBARC_PRINTF("the secret is:%s\r\n", public_data);
357  memset(public_data, 0, SECRET_LEN);
358  }
359  } else if (secret != NULL && new_pwd == NULL) {
360  if(container_call(container1, operate_secret, pwd, SET_SECRET, secret)) {
361  EMBARC_PRINTF("set new secret failed\r\n");
362  }
363  } else if (secret == NULL && new_pwd != NULL) {
364  if(container_call(container1, operate_secret, pwd, SET_PWD, new_pwd)) {
365  EMBARC_PRINTF("set new password failed\r\n");
366  }
367  } else {
368  EMBARC_PRINTF("cannot set password and secret at the same time\r\n");
369  }
370  } else {
371  EMBARC_PRINTF("no password input\r\n");
372  }
373 
374 error_exit:
375  return E_OK;
376 }
